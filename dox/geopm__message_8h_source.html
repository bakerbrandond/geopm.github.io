46 // Region id enums go from bit 63 and work their way down.
47 // Hint enums in geopm.h go from bit 32 and work their way up.
48 // There is a possibility of a conflict sometime in the future if they overlap.
49 enum geopm_region_id_e {
50  GEOPM_REGION_ID_EPOCH = 1ULL << 63, // Signaling the start of an epoch, no associated Region
51  GEOPM_REGION_ID_MPI = 1ULL << 62, // Execution of MPI calls
52  GEOPM_REGION_ID_UNMARKED = 1ULL << 61, // Code executed outside of a region
53  GEOPM_REGION_ID_UNDEFINED = 1ULL << 60, // Special value for an unset region id
54  GEOPM_NUM_REGION_ID_PRIVATE = 3, // Number table entries reserved for GEOPM defined regions (ignoring UNMARKED)
55 };

57 static inline int geopm_region_id_is_mpi(uint64_t rid)

67 static inline uint64_t geopm_region_id_hash(uint64_t rid)
68 {
69  if (rid != GEOPM_REGION_ID_EPOCH &&
70  rid != GEOPM_REGION_ID_UNMARKED) {
71  rid = ((rid << 32) >> 32);
72  }
73  return rid;
74 }
75 
76 static inline int geopm_region_id_is_nested(uint64_t rid)
77 {
78  return (geopm_region_id_is_mpi(rid) && geopm_region_id_hash(rid));
79 }
80 
81 static inline uint64_t geopm_region_id_parent(uint64_t rid)
82 {
83  return (geopm_region_id_is_nested(rid) ? geopm_region_id_hash(rid) : 0);
84 }
